// CONSTANTES DE TARIFAS const TARIFA_CARRO = 100; // Por minuto const TARIFA_MOTO = 50; // Por minuto const TOTAL_CUPOS = 20; // VALIDADOR Y RESCATE DE DATOS AL CARGAR LA PÁGINA (¡Aquí se soluciona el borrado!) let dbVehiculos = JSON.parse(localStorage.getItem('parqueadero_db')) || []; // Función para inicializar y pintar los datos guardados apenas se abre o actualiza la página function inicializarSistema() { actualizarVistaTablas(); actualizarContadorCupos(); } // REGISTRO DE ENTRADA (Módulo Taquilla) function registrarEntrada() { const placaInput = document.getElementById('placaIngreso').value.trim().toUpperCase(); const tipoVehiculo = document.getElementById('tipoVehiculo').value; // Validación 1: Campos vacíos if (!placaInput) { alert("🚨 ERROR: ¡La placa es obligatoria para registrar el ingreso!"); return; } // Validación 2: Longitud Máxima de Placa (¡Aquí arreglamos la Prueba 3!) if (placaInput.length > 6) { alert("🚨 ERROR: La placa no puede tener más de 6 caracteres (Formato Colombiano)."); return; } // Validación 3: Control de duplicados if (dbVehiculos.some(v => v.placa === placaInput)) { alert("⚠️ ATENCIÓN: Este vehículo ya se encuentra registrado dentro del parqueadero."); return; } // Validación 4: Control de cupo lleno if (dbVehiculos.length >= TOTAL_CUPOS) { alert("🛑 PARQUEADERO LLENO: No hay celdas disponibles en este momento."); return; } // Estructura del objeto vehículo con marca de tiempo real const nuevoVehiculo = { placa: placaInput, tipo: tipoVehiculo, horaIngreso: new Date().getTime() // Guarda milisegundos reales }; dbVehiculos.push(nuevoVehiculo); localStorage.setItem('parqueadero_db', JSON.stringify(dbVehiculos)); // Guarda en disco alert(`✅ ÉXITO: Vehículo ${placaInput} registrado correctamente.`); document.getElementById('placaIngreso').value = ""; // Limpia la caja actualizarVistaTablas(); actualizarContadorCupos(); } // PROCESAMIENTO DE SALIDA Y LIQUIDACIÓN MATEMÁTICA function registrarSalida() { const placaBuscar = document.getElementById('placaSalida').value.trim().toUpperCase(); if (!placaBuscar) { alert("🚨 ERROR: Ingrese una placa para procesar la salida."); return; } const vehiculoIndex = dbVehiculos.findIndex(v => v.placa === placaBuscar); if (vehiculoIndex === -1) { alert("❌ ERROR: El vehículo consultado no se encuentra activo en el parqueadero."); return; } const vehiculo = dbVehiculos[vehiculoIndex]; const horaSalida = new Date().getTime(); // Cálculo exacto de minutos transcurridos redondeando hacia arriba (Math.ceil) const diferenciaMilisegundos = horaSalida - vehiculo.horaIngreso; let minutos = Math.ceil(diferenciaMilisegundos / 60000); if (minutos <= 0) minutos = 1; // Tarifa mínima inicial de un minuto // Aplicación de tarifa por tipo const tarifaBase = (vehiculo.tipo === 'Carro') ? TARIFA_CARRO : TARIFA_MOTO; const totalPagar = minutos * tarifaBase; // Pintar el Ticket Digital en Pantalla de forma detallada const ticketHTML = `

*** RECIBO DE CAJA PARQUEADERO ***

FACTURA NO: AR-${Math.floor(Math.random() * 90000) + 10000}

PLACA: ${vehiculo.placa}

TIPO: ${vehiculo.tipo.toUpperCase()}

TIEMPO TOTAL: ${minutos} Minuto(s)

TOTAL A PAGAR: $${totalPagar} COP

¡Gracias por su confianza!

`; document.getElementById('zonaTicket').innerHTML = ticketHTML; // Eliminar del registro y actualizar persistencia dbVehiculos.splice(vehiculoIndex, 1); localStorage.setItem('parqueadero_db', JSON.stringify(dbVehiculos)); document.getElementById('placaSalida').value = ""; actualizarVistaTablas(); actualizarContadorCupos(); } // FUNCIONES AUXILIARES DE RENDERIZADO GRÁFICO function actualizarVistaTablas() { const tabla = document.getElementById('tablaVehiculos'); if (!tabla) return; tabla.innerHTML = ""; dbVehiculos.forEach(v => { const fecha = new Date(v.horaIngreso).toLocaleTimeString(); tabla.innerHTML += ` ${v.placa} ${v.tipo} ${fecha} `; }); } function actualizarContadorCupos() { const cuposDisponibles = TOTAL_CUPOS - dbVehiculos.length; const monitor = document.getElementById('contadorCupos'); if (monitor) { monitor.innerText = cuposDisponibles; if (cuposDisponibles === 0) { monitor.style.color = "red"; alert("🛑 ALERTA GLOBAL: Capacidad máxima agotada."); } else { monitor.style.color = "green"; } } } // ESCUCHADOR ASÍNCRONO ENTRE PESTAÑAS (Sincronización Móvil-Web en vivo) window.addEventListener('storage', (e) => { if (e.key === 'parqueadero_db') { dbVehiculos = JSON.parse(e.newValue) || []; actualizarVistaTablas(); actualizarContadorCupos(); } }); // DISPARADOR AUTOMÁTICO AL CARGAR O ACTUALIZAR LA PÁGINA window.onload = inicializarSistema;
